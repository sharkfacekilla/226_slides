def out_of_order(lst: [int]) -> [int]:

    new_lst = []

    if len(lst) < 2:

        return new_lst

    curr = lst[0]

    for i in lst[1:]:

        if curr <= i:

            new_lst.append(i)

        else:

            curr = i

    return new_lst

def create_asterisks(lst: [int]) -> [str]:

    new_lst = []

    for i in lst:

        if i > 0:

            new_lst.append('*' * i)

    return new_lst

def create_asterisks(lst: [int]) -> [str]:

    return ['*' * i for i in lst if i > 0]
